# 04 — 数据结构

## 一、list（列表）≈ JS Array

```python
fruits = ["apple", "banana", "cherry"]

# 增
fruits.append("date")            # 末尾添加
fruits.insert(1, "blueberry")   # 指定位置插入
fruits.extend(["fig", "grape"]) # 批量添加
fruits += ["fig", "grape"]      # 同上

# 删
fruits.pop()                     # 删除并返回最后一个
fruits.pop(0)                    # 删除并返回指定索引
fruits.remove("banana")          # 删除第一个匹配值
del fruits[1]                    # 删除指定索引
fruits.clear()                   # 清空

# 查
fruits[0]                        # 第一个
fruits[-1]                       # 最后一个（负索引）
fruits[1:3]                      # 切片 [start:stop)
fruits[::2]                      # 步长为 2
fruits[::-1]                     # 反转
len(fruits)                      # 长度
"apple" in fruits                # 成员检测 O(n)
fruits.index("apple")            # 查找索引
fruits.count("apple")            # 计数

# 改
fruits[0] = "avocado"

# 排序
fruits.sort()                    # 原地排序，返回 None！
fruits.sort(key=len)             # 按长度排序
sorted(fruits)                   # 返回新列表，不修改原列表
fruits.reverse()                 # 原地反转
```

```javascript
// JS 对比
const fruits = ["apple", "banana", "cherry"];
fruits.push("date");
fruits.splice(1, 0, "blueberry");
fruits.pop();
fruits.shift();
fruits.splice(1, 1);
fruits[0];
fruits.at(-1);          // 负索引
fruits.slice(1, 3);     // 切片
fruits.length;
fruits.includes("apple");
fruits.indexOf("apple");
fruits.sort();           // 返回数组本身（修改原数组）
[...fruits].sort();      // 不修改原数组
```

### 关键差异

| 操作 | Python | JS |
|------|--------|-----|
| 负索引 | `arr[-1]` 直接支持 | `arr.at(-1)` |
| 切片 | `arr[1:3]` 内置语法 | `arr.slice(1, 3)` |
| 排序返回 | `sort()` 返回 `None` | `sort()` 返回数组 |
| 步长切片 | `arr[::2]` | 无内置 |
| 成员检测 | `in` 关键字 | `.includes()` |

### 列表是引用类型

```python
a = [1, 2, 3]
b = a              # b 是 a 的引用（不是拷贝）
b.append(4)
print(a)           # [1, 2, 3, 4] —— a 也被修改了！

# 浅拷贝
b = a.copy()       # 或 b = a[:] 或 b = list(a)
b.append(5)
print(a)           # [1, 2, 3, 4] —— a 不受影响

# 深拷贝（嵌套对象需要）
import copy
b = copy.deepcopy(a)
```

## 二、dict（字典）≈ JS Object/Map

```python
user = {"name": "Alice", "age": 30}

# 增/改
user["email"] = "alice@example.com"
user.update({"phone": "123", "age": 31})  # 批量更新

# 删
del user["age"]
email = user.pop("email")         # 删除并返回
email = user.pop("email", None)   # 删除，不存在返回默认值
user.clear()

# 查
user["name"]                       # KeyError 如果不存在
user.get("name")                   # None 如果不存在
user.get("phone", "N/A")          # 指定默认值
"name" in user                     # 检查 key
list(user.keys())                  # 所有 key
list(user.values())                # 所有 value
list(user.items())                 # 所有 (key, value) 对
```

```javascript
// JS 对比
const user = { name: "Alice", age: 30 };
user.email = "alice@example.com";
Object.assign(user, { phone: "123", age: 31 });
delete user.age;
user.name;                  // 属性访问
user["name"];               // 括号访问
user.phone ?? "N/A";        // 空值合并
"name" in user;             // 检查属性
Object.keys(user);
Object.values(user);
Object.entries(user);
```

### dict 合并（Python 3.9+）

```python
a = {"x": 1, "y": 2}
b = {"y": 3, "z": 4}

# 合并运算符
c = a | b         # {'x': 1, 'y': 3, 'z': 4} —— b 覆盖 a 的同名 key

# 原地更新
a |= b            # a = a | b
```

### dict 推导式

```python
squares = {x: x**2 for x in range(5)}
# {0: 0, 1: 1, 2: 4, 3: 9, 4: 16}

# 反转字典
original = {"a": 1, "b": 2}
inverted = {v: k for k, v in original.items()}
# {1: 'a', 2: 'b'}
```

### defaultdict —— 带默认值的字典

```python
from collections import defaultdict

# 自动初始化不存在的 key
word_count = defaultdict(int)    # 默认值 0
for word in ["apple", "banana", "apple"]:
    word_count[word] += 1
# {'apple': 2, 'banana': 1}

# 分组
groups = defaultdict(list)
items = [("fruit", "apple"), ("veggie", "carrot"), ("fruit", "banana")]
for category, item in items:
    groups[category].append(item)
# {'fruit': ['apple', 'banana'], 'veggie': ['carrot']}
```

### Counter —— 计数器

```python
from collections import Counter

words = ["apple", "banana", "apple", "cherry", "banana", "apple"]
count = Counter(words)
# Counter({'apple': 3, 'banana': 2, 'cherry': 1})

count.most_common(2)  # [('apple', 3), ('banana', 2)]
count["apple"]        # 3
count["grape"]        # 0（不存在返回 0，不报错）
```

## 三、tuple（元组）—— 不可变序列

```python
# 元组是不可变的列表
point = (3, 4)
x, y = point          # 解包
# point[0] = 5        # TypeError: 不可修改

# 单元素元组需要逗号
single = (42,)         # 不是 (42) —— 那只是 42
not_tuple = (42)       # 这是整数 42

# 命名元组（更可读的元组）
from collections import namedtuple
Point = namedtuple("Point", ["x", "y"])
p = Point(3, 4)
print(p.x, p.y)       # 3 4

# 元组解包
a, b, *rest = (1, 2, 3, 4, 5)
# a=1, b=2, rest=[3, 4, 5]

first, *middle, last = (1, 2, 3, 4, 5)
# first=1, middle=[2, 3, 4], last=5
```

### 什么时候用 tuple

```python
# 1. 函数返回多个值
def get_user():
    return "Alice", 30     # 实际返回 tuple

name, age = get_user()     # 解包

# 2. 作为 dict 的 key（因为不可变，所以可哈希）
locations = {
    (40.7128, -74.0060): "NYC",
    (34.0522, -118.2437): "LA",
}

# 3. 不希望被修改的数据
WEEKDAYS = ("Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun")
```

## 四、set（集合）≈ JS Set

```python
s = {1, 2, 3, 3, 3}    # {1, 2, 3} —— 自动去重
s.add(4)
s.discard(2)            # 删除，不存在不报错
s.remove(1)             # 删除，不存在抛 KeyError

# 集合运算
a = {1, 2, 3}
b = {2, 3, 4}

a | b                   # 并集: {1, 2, 3, 4}
a & b                   # 交集: {2, 3}
a - b                   # 差集: {1}
a ^ b                   # 对称差集: {1, 4}

# 子集/超集
{1, 2} <= {1, 2, 3}     # True（子集）
{1, 2, 3} >= {1, 2}     # True（超集）
```

```javascript
// JS
const s = new Set([1, 2, 3, 3, 3]);
s.add(4);
s.delete(2);
s.has(1);
// JS Set 没有内置并集/交集运算（ES2025 新增）
```

### 集合推导式

```python
unique_lengths = {len(word) for word in ["hello", "world", "hi", "hey"]}
# {2, 5}
```

### 空集合陷阱

```python
# {} 是空 dict，不是空 set！
empty_dict = {}          # <class 'dict'>
empty_set = set()        # <class 'set'>
```

## 五、数据结构选择指南

| 需求 | 选择 | 原因 |
|------|------|------|
| 有序、可变、可重复 | `list` | 最通用 |
| 键值对 | `dict` | 快速查找 O(1) |
| 不可变序列 | `tuple` | 可哈希、轻量 |
| 去重、集合运算 | `set` | O(1) 查找 |
| 频率统计 | `Counter` | 专门用于计数 |
| 带默认值的字典 | `defaultdict` | 避免 KeyError |

## 六、collections 模块速查

```python
from collections import (
    defaultdict,   # 带默认值的 dict
    Counter,       # 计数器
    namedtuple,    # 命名元组
    deque,         # 双端队列（两头都能快速增删）
    OrderedDict,   # 有序字典（3.7+ dict 已有序，很少需要了）
)

# deque —— 双端队列
from collections import deque
dq = deque([1, 2, 3], maxlen=5)   # 最大长度 5
dq.appendleft(0)                   # 左端添加
dq.pop()                           # 右端弹出
dq.popleft()                       # 左端弹出
```

## 七、面试高频问题

**Q: list 和 tuple 的区别？**
A: list 可变、不可哈希；tuple 不可变、可哈希（可作为 dict key）。

**Q: dict 的时间复杂度？**
A: 查找、插入、删除都是 O(1)（平均），因为基于哈希表实现。

**Q: `sort()` 和 `sorted()` 的区别？**
A: `sort()` 原地排序、返回 `None`；`sorted()` 返回新列表、不修改原列表。`sorted()` 可用于任何可迭代对象。

**Q: 如何合并两个字典？**
A: `c = a | b`（Python 3.9+）或 `c = {**a, **b}`（3.5+）或 `a.update(b)`（原地）。
