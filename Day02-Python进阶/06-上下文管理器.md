# 06 — 上下文管理器

## 一、with 语句

```python
# 最常见 —— 文件操作
with open("data.txt", "r", encoding="utf-8") as f:
    content = f.read()
# 离开 with 块时自动关闭文件，即使发生异常

# 多个上下文管理器
with open("in.txt") as fin, open("out.txt", "w") as fout:
    fout.write(fin.read())
```

```javascript
// JS —— 没有直接等价物，用 try/finally
let f;
try {
    f = open("data.txt");
    const content = f.read();
} finally {
    f?.close();
}
```

## 二、自定义上下文管理器 —— 类方式

```python
class Timer:
    def __enter__(self):
        """进入 with 块时调用"""
        import time
        self.start = time.perf_counter()
        return self                # 赋给 as 变量

    def __exit__(self, exc_type, exc_val, exc_tb):
        """离开 with 块时调用"""
        import time
        self.elapsed = time.perf_counter() - self.start
        print(f"耗时: {self.elapsed:.4f}s")
        # 返回 True = 抑制异常
        # 返回 False/None = 让异常继续传播
        return False

with Timer() as t:
    sum(range(1000000))
print(f"实际耗时: {t.elapsed}s")
```

### `__exit__` 参数

```python
def __exit__(self, exc_type, exc_val, exc_tb):
    # exc_type:  异常类型（如 ValueError）
    # exc_val:   异常实例
    # exc_tb:    追踪对象
    # 无异常时三者都是 None

    if exc_type is not None:
        print(f"捕获到异常: {exc_type.__name__}: {exc_val}")
    return False  # 不抑制异常
```

## 三、@contextmanager —— 生成器方式

```python
from contextlib import contextmanager

@contextmanager
def timer():
    import time
    start = time.perf_counter()
    try:
        yield start    # yield 的值赋给 as 变量
    except Exception as e:
        print(f"异常: {e}")
        raise          # 重新抛出
    finally:
        elapsed = time.perf_counter() - start
        print(f"耗时: {elapsed:.4f}s")

with timer():
    sum(range(1000000))
```

### 关键点

```python
@contextmanager
def my_context():
    # yield 之前 = __enter__
    print("进入")
    resource = acquire_resource()

    try:
        yield resource     # resource 赋给 as 变量
    except Exception:
        # with 块中的异常在这里抛出
        print("处理异常")
        raise
    finally:
        # yield 之后 = __exit__
        print("离开")
        release_resource(resource)
```

## 四、实战：数据库连接

```python
from contextlib import contextmanager

@contextmanager
def db_connection(url: str):
    conn = connect(url)
    try:
        yield conn
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

with db_connection("postgresql://...") as conn:
    conn.execute("SELECT * FROM users")
```

## 五、实战：临时修改环境变量

```python
import os
from contextlib import contextmanager

@contextmanager
def env_var(key: str, value: str):
    old_value = os.environ.get(key)
    os.environ[key] = value
    try:
        yield
    finally:
        if old_value is None:
            os.environ.pop(key, None)
        else:
            os.environ[key] = old_value

with env_var("DEBUG", "true"):
    print(os.environ["DEBUG"])  # "true"
# DEBUG 恢复原值
```

## 六、suppress —— 忽略特定异常

```python
from contextlib import suppress

# 等价于 try/except + pass
with suppress(FileNotFoundError):
    os.remove("maybe_exists.txt")
# 文件不存在也不会报错
```

## 七、异步上下文管理器

```python
class AsyncDB:
    async def __aenter__(self):
        self.conn = await create_connection()
        return self.conn

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.conn.close()

async with AsyncDB() as conn:
    await conn.execute("...")
```

```python
# 异步 contextmanager
from contextlib import asynccontextmanager

@asynccontextmanager
async def async_timer():
    import asyncio
    start = asyncio.get_event_loop().time()
    try:
        yield
    finally:
        elapsed = asyncio.get_event_loop().time() - start
        print(f"异步耗时: {elapsed:.4f}s")
```

## 八、面试高频问题

**Q: `__enter__` 和 `__exit__` 分别什么时候调用？**
A: `__enter__` 在进入 `with` 块时调用，返回值赋给 `as` 变量。`__exit__` 在离开 `with` 块时调用（包括异常情况）。

**Q: `@contextmanager` 的 `yield` 发生了什么？**
A: `yield` 之前的代码是 `__enter__`，yield 的值赋给 `as` 变量。`yield` 之后的代码是 `__exit__`。`with` 块中的异常在 `yield` 处抛出。

**Q: 如何在 `__exit__` 中抑制异常？**
A: 返回 `True`。异常不会继续传播。通常不推荐，除非有充分理由。
