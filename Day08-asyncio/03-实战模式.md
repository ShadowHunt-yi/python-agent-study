# 03 — asyncio 实战模式

## 一、Agent 并发工具调用

```python
import asyncio
from openai import AsyncOpenAI

client = AsyncOpenAI()

async def execute_tools(tool_calls: list[dict]) -> list[str]:
    """并发执行多个工具调用"""
    async def execute_one(tc):
        name = tc["function"]["name"]
        args = tc["function"]["arguments"]
        # 实际执行工具
        return await run_tool(name, args)

    # 所有工具并发执行
    results = await asyncio.gather(
        *[execute_one(tc) for tc in tool_calls],
        return_exceptions=True,
    )

    # 处理异常
    processed = []
    for i, result in enumerate(results):
        if isinstance(result, Exception):
            processed.append(f"工具执行失败: {result}")
        else:
            processed.append(result)
    return processed
```

## 二、批量 LLM 调用

```python
import asyncio
from openai import AsyncOpenAI

client = AsyncOpenAI()

async def batch_call(queries: list[str], max_concurrent: int = 10) -> list[str]:
    """批量调用 LLM，控制并发数"""
    sem = asyncio.Semaphore(max_concurrent)

    async def call_one(query: str) -> str:
        async with sem:
            response = await client.chat.completions.create(
                model="gpt-4o",
                messages=[{"role": "user", "content": query}],
            )
            return response.choices[0].message.content

    results = await asyncio.gather(
        *[call_one(q) for q in queries],
        return_exceptions=True,
    )

    return [
        str(r) if not isinstance(r, Exception) else f"错误: {r}"
        for r in results
    ]

# 使用
queries = ["问题1", "问题2", "问题3", "问题4", "问题5"]
results = asyncio.run(batch_call(queries))
```

## 三、异步生成器 + 流式处理

```python
import asyncio

async def stream_chunks(text: str, chunk_size: int = 10):
    """将文本分块流式输出"""
    for i in range(0, len(text), chunk_size):
        await asyncio.sleep(0.05)  # 模拟延迟
        yield text[i:i+chunk_size]

async def main():
    async for chunk in stream_chunks("这是一段很长的文本..."):
        print(chunk, end="", flush=True)
```

## 四、异步上下文管理器

```python
import asyncio
from contextlib import asynccontextmanager

@asynccontextmanager
async def timed_operation(name: str):
    """计时上下文管理器"""
    start = asyncio.get_event_loop().time()
    try:
        yield
    finally:
        elapsed = asyncio.get_event_loop().time() - start
        print(f"{name} 耗时: {elapsed:.3f}s")

async def main():
    async with timed_operation("LLM 调用"):
        await asyncio.sleep(1)  # 模拟 LLM 调用
```

## 五、超时 + 重试组合

```python
import asyncio
from openai import AsyncOpenAI, RateLimitError, APIConnectionError

client = AsyncOpenAI()

async def call_with_retry_and_timeout(
    messages: list[dict],
    max_retries: int = 3,
    timeout: float = 30.0,
) -> str:
    """带重试和超时的 LLM 调用"""
    for attempt in range(max_retries):
        try:
            response = await asyncio.wait_for(
                client.chat.completions.create(
                    model="gpt-4o",
                    messages=messages,
                ),
                timeout=timeout,
            )
            return response.choices[0].message.content
        except asyncio.TimeoutError:
            if attempt == max_retries - 1:
                raise
            print(f"超时，重试 {attempt + 1}/{max_retries}")
        except (RateLimitError, APIConnectionError) as e:
            if attempt == max_retries - 1:
                raise
            wait = 2 ** attempt
            print(f"错误: {e}，{wait}s 后重试")
            await asyncio.sleep(wait)

    raise Exception("超过最大重试次数")
```

## 六、生产者-消费者模式

```python
import asyncio
from dataclasses import dataclass

@dataclass
class Task:
    id: int
    query: str

async def task_processor():
    """异步任务处理系统"""
    queue = asyncio.Queue(maxsize=100)
    results = {}

    async def producer():
        for i in range(20):
            await queue.put(Task(id=i, query=f"查询 {i}"))
        # 毒丸
        for _ in range(3):
            await queue.put(None)

    async def consumer(name: str):
        while True:
            task = await queue.get()
            if task is None:
                break
            # 模拟处理
            await asyncio.sleep(0.1)
            results[task.id] = f"{name} 处理了 {task.query}"
            queue.task_done()

    await asyncio.gather(
        producer(),
        consumer("C1"),
        consumer("C2"),
        consumer("C3"),
    )
    return results
```

## 七、异步 HTTP 批量请求

```python
import asyncio
import httpx

async def fetch_all(urls: list[str], max_concurrent: int = 10) -> list[dict]:
    """批量 HTTP 请求"""
    sem = asyncio.Semaphore(max_concurrent)

    async with httpx.AsyncClient(timeout=30.0) as client:
        async def fetch_one(url: str):
            async with sem:
                resp = await client.get(url)
                return {"url": url, "status": resp.status_code, "data": resp.json()}

        return await asyncio.gather(
            *[fetch_one(url) for url in urls],
            return_exceptions=True,
        )
```

## 八、面试高频问题

**Q: 怎么在 Agent 中并发执行工具？**
A: 用 `asyncio.gather()` 并发执行所有工具调用，用 `return_exceptions=True` 处理异常。

**Q: 怎么控制 LLM 调用的并发数？**
A: 用 `asyncio.Semaphore(n)` 限制同时进行的请求数。

**Q: 超时和重试怎么组合？**
A: `asyncio.wait_for()` 包裹调用实现超时，外层用 for 循环 + 指数退避实现重试。
