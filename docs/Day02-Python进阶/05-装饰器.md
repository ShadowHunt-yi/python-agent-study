# 05 — 装饰器

## 一、原理

装饰器就是一个接收函数、返回新函数的高阶函数。

```python
def my_decorator(func):
    def wrapper(*args, **kwargs):
        print("调用前")
        result = func(*args, **kwargs)
        print("调用后")
        return result
    return wrapper

# 手动应用
def say_hello(name):
    print(f"Hello, {name}")

say_hello = my_decorator(say_hello)

# @ 语法（等价）
@my_decorator
def say_hello(name):
    print(f"Hello, {name}")

say_hello("Alice")
# 调用前
# Hello, Alice
# 调用后
```

```javascript
// JS 对比 —— 高阶函数
const myDecorator = (fn) => (...args) => {
    console.log("调用前");
    const result = fn(...args);
    console.log("调用后");
    return result;
};

const sayHello = myDecorator((name) => console.log(`Hello, ${name}`));
```

## 二、functools.wraps —— 保留元信息

```python
import functools

def my_decorator(func):
    @functools.wraps(func)   # 保留原函数的 __name__, __doc__ 等
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def greet(name: str) -> str:
    """问候函数"""
    return f"Hello, {name}"

print(greet.__name__)   # "greet"（不是 "wrapper"）
print(greet.__doc__)    # "问候函数"
```

## 三、带参数的装饰器

```python
import functools

def repeat(num_times: int):
    """执行多次的装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for _ in range(num_times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(num_times=3)
def greet(name):
    print(f"Hello, {name}")

greet("Alice")
# Hello, Alice
# Hello, Alice
# Hello, Alice
```

### 三层嵌套结构

```python
# 带参数的装饰器 = 三层函数
def decorator_with_args(arg1, arg2):       # 1. 接收装饰器参数
    def decorator(func):                    # 2. 接收被装饰的函数
        @functools.wraps(func)
        def wrapper(*args, **kwargs):       # 3. 接收函数参数
            return func(*args, **kwargs)
        return wrapper
    return decorator
```

## 四、常用内置装饰器

```python
class MyClass:
    @property           # getter
    def name(self): ...

    @name.setter        # setter
    def name(self, value): ...

    @classmethod        # 类方法
    def from_string(cls, s): ...

    @staticmethod       # 静态方法
    def helper(): ...

# 缓存
@functools.lru_cache(maxsize=128)    # LRU 缓存
def fibonacci(n): ...

@functools.cache                      # 无限缓存（3.9+）
def expensive(x): ...

@functools.total_ordering             # 自动生成比较方法

class Point:
    x: float
    y: float
```

## 五、实战：计时装饰器

```python
import functools
import time

def timer(func):
    """打印函数执行时间"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} 耗时 {elapsed:.4f}s")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "done"

slow_function()   # slow_function 耗时 1.0012s
```

## 六、实战：重试装饰器

```python
import functools
import time

def retry(max_attempts: int = 3, delay: float = 1.0):
    """失败重试装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    print(f"尝试 {attempt} 失败: {e}, {delay}s 后重试...")
                    time.sleep(delay)
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def call_api():
    import random
    if random.random() < 0.7:
        raise ConnectionError("网络错误")
    return "成功"
```

## 七、实战：权限检查装饰器

```python
import functools

def require_auth(func):
    """检查用户是否已登录"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # 从 kwargs 或全局获取用户信息
        user = kwargs.get("current_user")
        if not user or not user.get("is_authenticated"):
            raise PermissionError("需要登录")
        return func(*args, **kwargs)
    return wrapper

@require_auth
def delete_user(user_id: int, current_user: dict = None):
    return f"Deleted user {user_id}"
```

## 八、类作为装饰器

```python
import functools

class CountCalls:
    """记录函数调用次数的装饰器类"""
    def __init__(self, func):
        functools.update_wrapper(self, func)
        self.func = func
        self.call_count = 0

    def __call__(self, *args, **kwargs):
        self.call_count += 1
        print(f"调用 #{self.call_count}: {self.func.__name__}")
        return self.func(*args, **kwargs)

@CountCalls
def process_data(data):
    return data * 2

process_data("test")      # 调用 #1: process_data
process_data("test")      # 调用 #2: process_data
process_data.call_count    # 2
```

## 九、多个装饰器叠加

```python
@decorator_a
@decorator_b
@decorator_c
def func():
    pass

# 等价于
func = decorator_a(decorator_b(decorator_c(func)))

# 执行顺序：decorator_c 先包装，decorator_a 最后包装
# 调用顺序：decorator_a 先执行，decorator_c 最后执行
```

```python
# 实际示例
@timer
@retry(max_attempts=3)
@require_auth
def sensitive_operation():
    pass

# 执行顺序：
# 1. timer 的 wrapper 开始计时
# 2. retry 的 wrapper 开始重试循环
# 3. require_auth 的 wrapper 检查权限
# 4. sensitive_operation 执行
```

## 十、异步装饰器

```python
import functools
import asyncio

def async_retry(max_attempts: int = 3):
    """异步重试装饰器 —— Agent 调用 LLM 必备"""
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    wait = 2 ** attempt  # 指数退避
                    print(f"尝试 {attempt} 失败: {e}, {wait}s 后重试...")
                    await asyncio.sleep(wait)
        return wrapper
    return decorator

@async_retry(max_attempts=3)
async def call_llm(client, messages):
    """调用 LLM，失败自动重试"""
    response = await client.chat.completions.create(
        model="gpt-4o", messages=messages,
    )
    return response.choices[0].message.content
```

### 异步 vs 同步装饰器的关键区别

```python
# ❌ 错误：用同步装饰器包装异步函数
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)  # 返回协程对象，没有 await！
    return wrapper

@bad_decorator
async def my_func():
    return 42

result = my_func()  # 返回 <coroutine object>，不是 42

# ✅ 正确：异步装饰器用 async def wrapper
def good_decorator(func):
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        return await func(*args, **kwargs)
    return wrapper
```

## 十一、Agent 开发中的常用装饰器

```python
import functools
import time
import logging

# 1. 工具日志装饰器 —— 记录每次工具调用
def log_tool_call(func):
    """记录工具调用的输入输出"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logging.info(f"调用工具 {func.__name__}: args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        logging.info(f"工具 {func.__name__} 返回: {result}")
        return result
    return wrapper

# 2. 工具超时装饰器
def tool_timeout(seconds: float = 30):
    """工具执行超时保护"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            import signal
            def handler(signum, frame):
                raise TimeoutError(f"工具 {func.__name__} 超时 ({seconds}s)")
            old_handler = signal.signal(signal.SIGALRM, handler)
            signal.alarm(int(seconds))
            try:
                return func(*args, **kwargs)
            finally:
                signal.alarm(0)
                signal.signal(signal.SIGALRM, old_handler)
        return wrapper
    return decorator

# 3. 缓存装饰器 —— 避免重复调用相同工具
from functools import lru_cache

def search_knowledge(query: str) -> str:
    """相同查询直接返回缓存结果"""
    return expensive_vector_search(query)
```

## 十二、装饰器叠加的执行顺序

```python
@timer
@retry(max_attempts=3)
@require_auth
def sensitive_operation():
    pass

# 等价于：sensitive_operation = timer(retry(require_auth(sensitive_operation)))
# 定义时执行顺序（从下到上）：require_auth → retry → timer
# 调用时执行顺序（从外到内）：timer → retry → require_auth → sensitive_operation

# 调用流程图：
# timer.wrapper 开始计时
#   → retry.wrapper 开始重试循环
#     → require_auth.wrapper 检查权限
#       → sensitive_operation 执行
#     ← require_auth 返回
#   ← retry 返回（或重试）
# ← timer 输出耗时
```

## 十三、面试高频问题

**Q: 装饰器的本质是什么？**
A: 一个接收函数作为参数、返回新函数的高阶函数。`@decorator` 是 `func = decorator(func)` 的语法糖。

**Q: 为什么需要 `functools.wraps`？**
A: 保留被装饰函数的元信息（`__name__`、`__doc__`、`__module__` 等）。不加的话，调试时看到的函数名是 `wrapper` 而不是原函数名。

**Q: 带参数的装饰器和不带参数的有什么区别？**
A: 不带参数的是两层函数（装饰器+wrapper）。带参数的是三层函数（参数接收器+装饰器+wrapper）。

**Q: 装饰器的执行时机？**
A: 在模块加载时执行（定义时），不是调用时。`@decorator` 在 `def` 语句执行时立即应用。

**Q: 异步函数能用同步装饰器吗？**
A: 不能。同步装饰器的 `wrapper` 返回的是协程对象而非结果。异步函数必须用 `async def wrapper` + `await func()` 的装饰器。

**Q: 多个装饰器叠加的执行顺序？**
A: 定义时从下到上（内到外），调用时从上到下（外到内）。`@A @B @C def f` 等价于 `f = A(B(C(f)))`，调用时 A 先执行。
