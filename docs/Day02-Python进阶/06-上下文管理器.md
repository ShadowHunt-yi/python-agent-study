# 06 — 上下文管理器

## 一、with 语句

```python
# 最常见 —— 文件操作
with open("data.txt", "r", encoding="utf-8") as f:
    content = f.read()
# 离开 with 块时自动关闭文件，即使发生异常

# 多个上下文管理器
with open("in.txt") as fin, open("out.txt", "w") as fout:
    fout.write(fin.read())
```

```javascript
// JS —— 没有直接等价物，用 try/finally
let f;
try {
    f = open("data.txt");
    const content = f.read();
} finally {
    f?.close();
}
```

## 二、自定义上下文管理器 —— 类方式

```python
class Timer:
    def __enter__(self):
        """进入 with 块时调用"""
        import time
        self.start = time.perf_counter()
        return self                # 赋给 as 变量

    def __exit__(self, exc_type, exc_val, exc_tb):
        """离开 with 块时调用"""
        import time
        self.elapsed = time.perf_counter() - self.start
        print(f"耗时: {self.elapsed:.4f}s")
        # 返回 True = 抑制异常
        # 返回 False/None = 让异常继续传播
        return False

with Timer() as t:
    sum(range(1000000))
print(f"实际耗时: {t.elapsed}s")
```

### `__exit__` 参数

```python
def __exit__(self, exc_type, exc_val, exc_tb):
    # exc_type:  异常类型（如 ValueError）
    # exc_val:   异常实例
    # exc_tb:    追踪对象
    # 无异常时三者都是 None

    if exc_type is not None:
        print(f"捕获到异常: {exc_type.__name__}: {exc_val}")
    return False  # 不抑制异常
```

## 三、@contextmanager —— 生成器方式

```python
from contextlib import contextmanager

@contextmanager
def timer():
    import time
    start = time.perf_counter()
    try:
        yield start    # yield 的值赋给 as 变量
    except Exception as e:
        print(f"异常: {e}")
        raise          # 重新抛出
    finally:
        elapsed = time.perf_counter() - start
        print(f"耗时: {elapsed:.4f}s")

with timer():
    sum(range(1000000))
```

### 关键点

```python
@contextmanager
def my_context():
    # yield 之前 = __enter__
    print("进入")
    resource = acquire_resource()

    try:
        yield resource     # resource 赋给 as 变量
    except Exception:
        # with 块中的异常在这里抛出
        print("处理异常")
        raise
    finally:
        # yield 之后 = __exit__
        print("离开")
        release_resource(resource)
```

## 四、实战：数据库连接

```python
from contextlib import contextmanager

@contextmanager
def db_connection(url: str):
    conn = connect(url)
    try:
        yield conn
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

with db_connection("postgresql://...") as conn:
    conn.execute("SELECT * FROM users")
```

## 五、实战：临时修改环境变量

```python
import os
from contextlib import contextmanager

@contextmanager
def env_var(key: str, value: str):
    old_value = os.environ.get(key)
    os.environ[key] = value
    try:
        yield
    finally:
        if old_value is None:
            os.environ.pop(key, None)
        else:
            os.environ[key] = old_value

with env_var("DEBUG", "true"):
    print(os.environ["DEBUG"])  # "true"
# DEBUG 恢复原值
```

## 八、__exit__ 异常处理详解

```python
class SafeFileWriter:
    def __init__(self, path):
        self.path = path

    def __enter__(self):
        self.f = open(self.path, "w")
        return self.f

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.f.close()

        if exc_type is not None:
            print(f"写入时出错: {exc_type.__name__}: {exc_val}")

        # 返回 True = 抑制异常（with 块外看不到异常）
        # 返回 False/None = 异常继续传播（默认行为）
        return True  # 抑制异常

# 使用：即使 with 块内抛异常，外面也不会崩溃
with SafeFileWriter("out.txt") as f:
    f.write("hello")
    raise ValueError("出错了")  # 被 __exit__ 抑制

print("程序继续运行")  # 会执行
```

### 什么时候抑制异常？

```python
# ✅ 合理场景：事务回滚后抑制异常
class Transaction:
    def __exit__(self, exc_type, exc_val, exc_tb):
        if exc_type:
            self.conn.rollback()  # 回滚事务
            return True           # 抑制异常，让调用方决定是否重试
        self.conn.commit()

# ❌ 不合理场景：盲目抑制所有异常
class BadContext:
    def __exit__(self, *args):
        return True  # 永远抑制异常，隐藏 bug
```

## 九、FastAPI Lifespan —— 应用生命周期

```python
from contextlib import asynccontextmanager
from fastapi import FastAPI

@asynccontextmanager
async def lifespan(app: FastAPI):
    # 启动时执行（类似 __enter__）
    print("应用启动")
    app.state.db = await create_db_pool()
    app.state.llm_client = AsyncOpenAI()

    yield  # 应用运行期间

    # 关闭时执行（类似 __exit__）
    print("应用关闭")
    await app.state.db.close()
    await app.state.llm_client.close()

app = FastAPI(lifespan=lifespan)

# 在路由中使用启动时创建的资源
@app.get("/chat")
async def chat(msg: str):
    client = app.state.llm_client
    response = await client.chat.completions.create(...)
    return response
```

### Agent 中的上下文管理器

```python
from contextlib import contextmanager

@contextmanager
def agent_trace(trace_id: str):
    """Agent 执行追踪上下文"""
    start = time.time()
    trace = {"id": trace_id, "steps": [], "status": "running"}
    try:
        yield trace
        trace["status"] = "success"
    except Exception as e:
        trace["status"] = "error"
        trace["error"] = str(e)
        raise
    finally:
        trace["duration"] = time.time() - start
        save_trace(trace)  # 持久化追踪数据

# 使用
with agent_trace("trace-001") as trace:
    trace["steps"].append({"type": "llm_call", "model": "gpt-4o"})
    result = await call_llm(messages)
    trace["steps"].append({"type": "tool_call", "tool": "search"})
```

## 十、面试高频问题

**Q: `__enter__` 和 `__exit__` 分别什么时候调用？**
A: `__enter__` 在进入 `with` 块时调用，返回值赋给 `as` 变量。`__exit__` 在离开 `with` 块时调用（包括异常情况）。

**Q: `@contextmanager` 的 `yield` 发生了什么？**
A: `yield` 之前的代码是 `__enter__`，yield 的值赋给 `as` 变量。`yield` 之后的代码是 `__exit__`。`with` 块中的异常在 `yield` 处抛出。

**Q: 如何在 `__exit__` 中抑制异常？**
A: 返回 `True`。异常不会继续传播。通常不推荐，除非有充分理由（如事务回滚）。

**Q: `@contextmanager` 和类方式哪个更好？**
A: 简单场景用 `@contextmanager`（代码更少，逻辑集中）。复杂场景用类（可以保存状态、有多个方法）。Agent 开发中 `@contextmanager` 更常用。

**Q: FastAPI 的 `lifespan` 是什么？**
A: 应用级上下文管理器，`yield` 前是启动逻辑（创建连接池、初始化客户端），`yield` 后是关闭逻辑（释放资源）。替代了旧的 `@app.on_event("startup")`。
