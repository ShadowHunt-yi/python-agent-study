# 10 — 数据处理基础

## 一、collections 模块

```python
from collections import defaultdict, Counter, deque, namedtuple

# defaultdict —— 带默认值的字典
word_count = defaultdict(int)
for word in ["a", "b", "a", "c", "a"]:
    word_count[word] += 1
# {'a': 3, 'b': 1, 'c': 1}

# Counter —— 计数器
counter = Counter(["a", "b", "a", "c", "a"])
counter.most_common(2)   # [('a', 3), ('b', 1)]

# deque —— 双端队列
dq = deque(maxlen=3)
dq.append(1)
dq.append(2)
dq.append(3)
dq.append(4)   # 自动丢弃左边
# deque([2, 3, 4], maxlen=3)

# namedtuple —— 命名元组
Point = namedtuple("Point", ["x", "y"])
p = Point(1, 2)
print(p.x, p.y)   # 1 2
```

## 二、itertools

```python
import itertools

# chain —— 链接多个迭代器
list(itertools.chain([1, 2], [3, 4]))   # [1, 2, 3, 4]

# islice —— 切片生成器
list(itertools.islice(range(100), 5))   # [0, 1, 2, 3, 4]

# groupby —— 分组
data = [("a", 1), ("a", 2), ("b", 3)]
for key, group in itertools.groupby(data, key=lambda x: x[0]):
    print(key, list(group))

# product —— 笛卡尔积
list(itertools.product([1, 2], ["a", "b"]))
# [(1, 'a'), (1, 'b'), (2, 'a'), (2, 'b')]

# combinations —— 组合
list(itertools.combinations([1, 2, 3], 2))
# [(1, 2), (1, 3), (2, 3)]

# permutations —— 排列
list(itertools.permutations([1, 2, 3], 2))
# [(1, 2), (1, 3), (2, 1), (2, 3), (3, 1), (3, 2)]
```

## 三、functools

```python
import functools

# lru_cache —— 缓存
@functools.lru_cache(maxsize=128)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# partial —— 偏函数
def power(base, exp):
    return base ** exp

square = functools.partial(power, exp=2)
cube = functools.partial(power, exp=3)
square(5)   # 25
cube(3)     # 27

# reduce —— 归约
functools.reduce(lambda a, b: a + b, [1, 2, 3, 4])   # 10

# total_ordering —— 自动生成比较方法
@functools.total_ordering
class Student:
    def __init__(self, grade):
        self.grade = grade
    def __eq__(self, other):
        return self.grade == other.grade
    def __lt__(self, other):
        return self.grade < other.grade
# 自动生成 __le__, __gt__, __ge__
```

## 四、正则表达式

```python
import re

text = "我的邮箱是 alice@example.com，电话是 13812345678"

# 查找
email = re.search(r"[\w.-]+@[\w.-]+", text)
if email:
    print(email.group())   # alice@example.com

# 查找所有
phones = re.findall(r"1[3-9]\d{9}", text)
print(phones)   # ['13812345678']

# 替换
masked = re.sub(r"(1[3-9]\d{2})\d{4}(\d{4})", r"\1****\2", text)
print(masked)   # 138****5678

# 分割
parts = re.split(r"[,，。.]", "你好，世界。再见.")
print(parts)   # ['你好', '世界', '再见', '']
```

## 五、日期时间

```python
from datetime import datetime, timedelta

now = datetime.now()
print(now.strftime("%Y-%m-%d %H:%M:%S"))   # 2026-05-20 10:30:00

# 解析
dt = datetime.strptime("2026-05-20", "%Y-%m-%d")

# 计算
tomorrow = now + timedelta(days=1)
diff = datetime(2026, 12, 31) - now
print(diff.days)   # 剩余天数
```

## 七、面试高频问题

**Q: defaultdict 和普通 dict 的区别？**
A: defaultdict 访问不存在的 key 时自动创建默认值。普通 dict 会抛 KeyError。

**Q: lru_cache 的作用？**
A: 缓存函数结果，相同参数不重复计算。适合递归或耗时计算。

**Q: 怎么处理日期时间？**
A: 用 datetime 模块。格式化用 strftime，解析用 strptime。时区处理用 zoneinfo（3.9+）。
