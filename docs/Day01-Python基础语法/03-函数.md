# 03 — 函数

## 一、基础定义

```python
# Python
def greet(name):
    return f"Hello, {name}"  # f"..." 格式化字符串，{name} 会被替换为变量值，详见 05-字符串格式化.md

def add(a, b):
    return a + b

# 没有 return 时隐式返回 None
def log(message):
    print(message)
    # 返回 None
```

```javascript
// JS 对比
function greet(name) {
    return `Hello, ${name}`;
}

const add = (a, b) => a + b;

function log(message) {
    console.log(message);
    // 返回 undefined
}
```

### 关键差异

| 差异 | Python | JS |
|------|--------|-----|
| 关键字 | `def` | `function` / `=>` |
| 无返回值 | 返回 `None` | 返回 `undefined` |
| 箭头函数 | 无（用 `lambda` 替代，限单表达式） | `() => {}` |
| 函数是一等公民 | 是 | 是 |

## 二、默认参数

```python
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}"

greet("Alice")                  # → "Hello, Alice"
greet("Alice", "Hi")            # → "Hi, Alice"
greet("Alice", greeting="Hey")  # → "Hey, Alice" —— 关键字参数
```

```javascript
// JS
function greet(name, greeting = "Hello") {
    return `${greeting}, ${name}`;
}
```

### 可变默认参数陷阱（高频面试题）

```python
# ✗ 危险 —— 默认值在函数定义时只计算一次
def append_to(item, target=[]):
    target.append(item)
    return target

append_to(1)   # [1]
append_to(2)   # [1, 2] —— 不是 [2]！列表被共享了

# ✓ 正确写法
def append_to(item, target=None):
    if target is None:
        target = []
    target.append(item)
    return target

append_to(1)   # [1]
append_to(2)   # [2] —— 每次都是新列表
```

**原理**：Python 的默认参数在 `def` 语句执行时求值一次，之后所有调用共享同一个对象。JS 的默认参数在每次调用时重新求值。

## 三、*args 和 **kwargs

```python
# *args —— 可变位置参数（收集为 tuple）
def sum_all(*args):
    print(type(args))  # <class 'tuple'>
    return sum(args)

sum_all(1, 2, 3)       # 6

# **kwargs —— 可变关键字参数（收集为 dict）
def print_info(**kwargs):
    print(type(kwargs))  # <class 'dict'>
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Alice", age=30)
# → name: Alice
# → age: 30

# 混合使用（顺序必须：位置参数, *args, 关键字参数, **kwargs）
def func(a, b, *args, **kwargs):
    print(f"a={a}, b={b}, args={args}, kwargs={kwargs}")

func(1, 2, 3, 4, x=5, y=6)
# → a=1, b=2, args=(3, 4), kwargs={'x': 5, 'y': 6}
```

```javascript
// JS 对比
function sumAll(...args) {
    return args.reduce((a, b) => a + b, 0);
}

function printInfo(options) {
    for (const [key, value] of Object.entries(options)) {
        console.log(`${key}: ${value}`);
    }
}
```

### 参数解包

```python
def add(a, b, c):
    return a + b + c

numbers = [1, 2, 3]
add(*numbers)       # 6 —— 等价于 add(1, 2, 3)

user = {"a": 1, "b": 2, "c": 3}
add(**user)          # 6 —— 等价于 add(a=1, b=2, c=3)
```

## 四、lambda 函数

```python
# Python lambda —— 限单表达式
square = lambda x: x ** 2
add = lambda a, b: a + b

# 常见用途：排序、map、filter
numbers = [3, 1, 4, 1, 5]
sorted(numbers, key=lambda x: -x)          # [5, 4, 3, 1, 1]
list(map(lambda x: x * 2, numbers))        # [6, 2, 8, 2, 10]
list(filter(lambda x: x > 2, numbers))     # [3, 4, 5]
```

```javascript
// JS
const square = (x) => x ** 2;
const add = (a, b) => a + b;

numbers.sort((a, b) => b - a);
numbers.map(x => x * 2);
numbers.filter(x => x > 2);
```

### lambda 限制

```python
# lambda 只能是单表达式，不能有语句
# ✗ 不能这样
lambda x: if x > 0: return x

# ✓ 用普通函数
def positive(x):
    if x > 0:
        return x
    return 0
```

## 五、类型提示（Type Hints）

类型提示不强制执行，仅用于静态分析和 IDE 提示。

```python
# 基础类型提示
def greet(name: str) -> str:
    return f"Hello, {name}"

def add(a: int, b: int = 0) -> int:
    return a + b

# 复杂类型
from typing import Optional, Union, List, Dict

def find_user(user_id: int) -> Optional[dict]:
    """返回用户字典或 None"""
    ...

def process(value: Union[str, int]) -> str:
    return str(value)

# Python 3.10+ 用 | 替代 Union
def process(value: str | int) -> str:
    return str(value)

# 容器类型
def get_names() -> list[str]:
    return ["Alice", "Bob"]

def get_scores() -> dict[str, int]:
    return {"Alice": 95, "Bob": 87}

# 类型别名
from typing import TypeAlias
Vector: TypeAlias = list[float]
Matrix: TypeAlias = list[Vector]
```

### Callable 类型

```python
from typing import Callable

# 函数作为参数
def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
    return func(a, b)

apply(add, 3, 4)  # 7
```

### TypeVar —— 泛型

```python
from typing import TypeVar

T = TypeVar("T")

def first(items: list[T]) -> T | None:
    return items[0] if items else None

first([1, 2, 3])     # int: 1
first(["a", "b"])    # str: "a"
```

## 六、作用域（LEGB）

```python
# Python 的变量查找顺序：Local → Enclosing → Global → Built-in

x = "global"

def outer():
    x = "enclosing"
    
    def inner():
        x = "local"
        print(x)     # "local"
    
    inner()
    print(x)         # "enclosing"

outer()
print(x)             # "global"
```

### global 和 nonlocal

```python
count = 0

def increment():
    global count          # 声明使用全局变量
    count += 1

increment()
print(count)  # 1


def outer():
    x = 10
    
    def inner():
        nonlocal x        # 声明使用外层变量
        x += 1
    
    inner()
    print(x)  # 11

outer()
```

## 七、文档字符串（Docstring）

```python
def calculate_bmi(weight: float, height: float) -> float:
    """计算 BMI 指数。

    Args:
        weight: 体重（千克）
        height: 身高（米）

    Returns:
        BMI 值

    Raises:
        ValueError: 身高为 0 时
    """
    if height <= 0:
        raise ValueError("身高必须大于 0")
    return weight / (height ** 2)

# 访问文档
help(calculate_bmi)
calculate_bmi.__doc__
```

## 八、函数式编程工具

```python
# map —— 对每个元素应用函数
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x ** 2, numbers))
# [1, 4, 9, 16, 25]

# filter —— 过滤
evens = list(filter(lambda x: x % 2 == 0, numbers))
# [2, 4]

# reduce —— 累积
from functools import reduce
total = reduce(lambda acc, x: acc + x, numbers, 0)
# 15

# sorted —— 返回新列表（不修改原列表）
sorted_numbers = sorted(numbers, reverse=True)
# [5, 4, 3, 2, 1]
```

```javascript
// JS 对比
const squared = numbers.map(x => x ** 2);
const evens = numbers.filter(x => x % 2 === 0);
const total = numbers.reduce((acc, x) => acc + x, 0);
const sorted = [...numbers].sort((a, b) => b - a);
```

## 九、面试高频问题

**Q: `*args` 和 `**kwargs` 是什么？**
A: `*args` 收集多余的位置参数为 tuple，`**kwargs` 收集多余的关键字参数为 dict。命名可以自定义（如 `*numbers`），但 `*` 和 `**` 是语法要求。

**Q: 可变默认参数为什么危险？**
A: 默认值在 `def` 时只计算一次，所有调用共享同一个对象。应该用 `None` 作为默认值，在函数体内初始化。

**Q: Python 有箭头函数吗？**
A: 有 `lambda`，但限制为单表达式。复杂逻辑必须用 `def`。

**Q: 类型提示会影响运行时性能吗？**
A: 不会。类型提示在运行时被忽略（除非用 Pydantic 等框架主动读取）。它们只用于静态分析工具（mypy）和 IDE 提示。
