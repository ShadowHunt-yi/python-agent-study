# 04 — Pydantic 高级用法

## 一、自定义类型

```python
from pydantic import BaseModel, Field, field_validator
from typing import Annotated

# 用 Annotated 组合约束
PositiveInt = Annotated[int, Field(gt=0)]
NonEmptyStr = Annotated[str, Field(min_length=1)]
EmailStr = Annotated[str, Field(pattern=r"^[\w.-]+@[\w.-]+\.\w+$")]

class User(BaseModel):
    id: PositiveInt
    name: NonEmptyStr
    email: EmailStr

# 自定义类型别名
UserID = Annotated[int, Field(gt=0, description="用户唯一标识")]
```

## 二、泛型模型

```python
from pydantic import BaseModel
from typing import TypeVar, Generic

T = TypeVar("T")

class Response(BaseModel, Generic[T]):
    code: int = 200
    message: str = "success"
    data: T | None = None

# 使用
class UserData(BaseModel):
    name: str
    age: int

resp = Response[UserData](data=UserData(name="Alice", age=30))
print(resp.model_dump())
# {'code': 200, 'message': 'success', 'data': {'name': 'Alice', 'age': 30}}
```

## 三、Discriminated Unions（鉴别联合）

```python
from pydantic import BaseModel, Field
from typing import Literal, Union

class TextContent(BaseModel):
    type: Literal["text"] = "text"
    text: str

class ImageContent(BaseModel):
    type: Literal["image"] = "image"
    url: str
    width: int

class Message(BaseModel):
    content: TextContent | ImageContent = Field(discriminator="type")

# 自动根据 type 字段选择正确的模型
msg1 = Message(content={"type": "text", "text": "hello"})
msg2 = Message(content={"type": "image", "url": "http://...", "width": 100})
```

### Agent 消息类型

```python
from pydantic import BaseModel, Field
from typing import Literal

class UserMessage(BaseModel):
    role: Literal["user"] = "user"
    content: str

class AssistantMessage(BaseModel):
    role: Literal["assistant"] = "assistant"
    content: str
    tool_calls: list[dict] = []

class ToolMessage(BaseModel):
    role: Literal["tool"] = "tool"
    content: str
    tool_call_id: str

class SystemMessage(BaseModel):
    role: Literal["system"] = "system"
    content: str

ChatMessage = UserMessage | AssistantMessage | ToolMessage | SystemMessage
```

## 四、序列化控制

```python
from pydantic import BaseModel, Field, ConfigDict
from datetime import datetime

class Record(BaseModel):
    model_config = ConfigDict(serialize_by_alias=True)

    id: int = Field(alias="recordId")
    created_at: datetime = Field(alias="createdAt")
    data: dict = Field(default_factory=dict)

    # 自定义序列化器
    @field_serializer("created_at")
    def serialize_dt(self, dt: datetime, _info):
        return dt.strftime("%Y-%m-%d %H:%M:%S")

record = Record(recordId=1, createdAt=datetime.now())
print(record.model_dump(by_alias=True))
# {'recordId': 1, 'createdAt': '2026-05-20 10:30:00', 'data': {}}
```

## 五、model_post_init 和 model_validator(mode="wrap")

```python
from pydantic import BaseModel, model_validator

class Config(BaseModel):
    host: str
    port: int
    url: str = ""

    @model_validator(mode="after")
    def build_url(self) -> "Config":
        if not self.url:
            self.url = f"http://{self.host}:{self.port}"
        return self

config = Config(host="localhost", port=8000)
print(config.url)  # "http://localhost:8000"
```

## 六、递归模型

```python
from pydantic import BaseModel

class TreeNode(BaseModel):
    value: int
    children: list["TreeNode"] = []

tree = TreeNode(
    value=1,
    children=[
        TreeNode(value=2, children=[TreeNode(value=4)]),
        TreeNode(value=3),
    ],
)
print(tree.model_dump())
```

## 七、Agent 实战：工具参数 Schema

```python
from pydantic import BaseModel, Field
from openai import OpenAI

class SearchInput(BaseModel):
    """搜索互联网获取信息"""
    query: str = Field(description="搜索关键词")
    max_results: int = Field(default=5, ge=1, le=20, description="最大结果数")

class CalculatorInput(BaseModel):
    """计算数学表达式"""
    expression: str = Field(description="数学表达式")

# 直接生成 OpenAI tools 格式
def make_tool(model: BaseModel, name: str) -> dict:
    return {
        "type": "function",
        "function": {
            "name": name,
            "description": model.__doc__,
            "parameters": model.model_json_schema(),
        },
    }

tools = [
    make_tool(SearchInput, "search"),
    make_tool(CalculatorInput, "calculate"),
]

client = OpenAI()
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "搜一下 Python 3.12 新特性"}],
    tools=tools,
)
```

## 八、面试高频问题

**Q: Discriminated Unions 是什么？**
A: 用一个字段（如 `type`）区分不同的子模型。Pydantic 根据该字段自动选择正确的模型进行解析。

**Q: 泛型模型怎么用？**
A: `class Response(BaseModel, Generic[T])` 定义泛型，`Response[User]` 使用。

**Q: 怎么在 Pydantic 中处理递归数据？**
A: 直接引用自身类型（`list["TreeNode"]`），Pydantic 自动处理。
